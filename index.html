        function addFood() {

            const name = document.getElementById('foodName').value;

            const date = document.getElementById('expiryDate').value;

            const loc = document.getElementById('storageLocation').value;

            if(!name || !date) return alert("Remplissez le nom et la date !");

            inventory.push({ name, date, loc, selected: false });

            save(); render();

            document.getElementById('foodName').value = '';

        }

        function deleteFood(idx) { inventory.splice(idx, 1); save(); render(); }

        function save() { localStorage.setItem('inventory', JSON.stringify(inventory)); }

       

        function isUrgent(dateStr) {

            const today = new Date();

            today.setHours(0,0,0,0);

            const expiry = new Date(dateStr);

            expiry.setHours(0,0,0,0);

            return (expiry - today) / (1000 * 60 * 60 * 24) <= 2;

        }

        function render() {

            inventory.sort((a, b) => new Date(a.date) - new Date(b.date));

            document.getElementById('productList').innerHTML = inventory.map((i, idx) => {

                const urgent = isUrgent(i.date);

                return `

                <div class="item ${urgent ? 'urgent' : ''} ${i.selected ? 'selected' : ''}">

                    <div class="checkbox-container">

                        <input type="checkbox" ${i.selected ? 'checked' : ''} onchange="inventory[${idx}].selected = this.checked; save(); render();">

                    </div>

                    <span class="visual">${getVisual(i.name)}</span>

                    <div class="info">

                        <div class="item-title">${i.name}</div>

                        <div class="item-meta">${i.date} • ${i.loc}</div>

                        ${urgent ? '<span class="alert-text">⚠️ Consommer sous 2 jours</span>' : ''}

                    </div>

                    <button class="del-btn" onclick="deleteFood(${idx})">Supprimer</button>

                </div>

            `}).join('');

        }

        function searchGoogle() {

            const selected = inventory.filter(i => i.selected).map(i => i.name).join(' ');

            if(!selected) return alert("Sélectionnez au moins un ingrédient !");

            window.open(`https://www.google.com/search?q=${encodeURIComponent(selected + " recette anti-gaspi")}`, '_blank');

        }
